# TeslaUSB Neo - Raspberry Pi Zero 2 W boot config

# GPU firmware. MUST be set explicitly: with gpu_mem=16 below, the boot ROM
# would otherwise auto-select the cut-down start_cd.elf/fixup_cd.dat, which
# Buildroot's BR2_PACKAGE_RPI_FIRMWARE_VARIANT_PI does NOT install — only
# start.elf/fixup.dat are present. Without these two lines the Pi hangs in
# the firmware stage and never boots. (Matches upstream config_zero2w.txt.)
start_file=start.elf
fixup_file=fixup.dat

# Kernel
kernel=zImage
arm_64bit=0

# USB OTG gadget mode. dr_mode=otg matches the Pi OS path (setup.sh) so both
# images behave identically when plugged into the car (a powered host).
dtoverlay=dwc2,dr_mode=otg

# Disable HDMI (headless, saves ~25mA and boot time)
hdmi_blanking=2

# Disable audio (not needed)
dtparam=audio=off

# Minimize GPU memory (headless)
gpu_mem=16

# Enable hardware watchdog
dtparam=watchdog=on

# Disable HAT/PoE probe (saves ~200ms boot time)
force_eeprom_read=0

# Disable camera auto-detect (saves boot time)
camera_auto_detect=0

# Disable display auto-detect (saves boot time)
display_auto_detect=0
